# vueUI
the UI library of vue

## install
```js
npm install @hero_666/web_vueui
```

## start
```js
//  import
import vueUI from '@hero_666/web_vueui';
import '@hero_666/web_vueui/dist/vueUI.css';
import Vue from 'vue';

Vue.use(vueUI,{
    errorImg: 'data:image/png;base64.......' // the base64 data of image
})


```
